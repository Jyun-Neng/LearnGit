git command:
	mkdir: create a new file
	git init: initialized empty repository
	git status: show the working tree
	git remote add origin URL: connect to the repository
	git add: add file
	git commit -m "description": commit the change
	git push origin master: push to master
	git push origin --delete branch_name: delete the branch_name from github	
	git config remote.origin.url: show the url
	git remote set-url origin URL: change the url
	
	git clone URL: create the github file to the pc

	git checkout branch_name: switch to the branch_name
	git checkout -b branch_name: create a new branch
	git branch: show all the branches
	git branch -d branch_name: delete the branch_name from PC
	git merge branch_name: merge the branch_name to master
